Setup instructions
Run this SQL in your Supabase SQL editor to create the tables
CREATE TABLE IF NOT EXISTS primus_logs (
  id BIGSERIAL PRIMARY KEY,
  log_date DATE NOT NULL UNIQUE,
  diet_pct INTEGER, workout_pct INTEGER,
  water_glasses INTEGER, calories_eaten INTEGER,
  protein_g INTEGER, carbs_g INTEGER, fat_g INTEGER,
  meals_done INTEGER, meals_total INTEGER,
  exercises_done INTEGER, exercises_total INTEGER,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS primus_journal (
  id BIGSERIAL PRIMARY KEY,
  entry_date DATE NOT NULL,
  mood TEXT, note_text TEXT, tags TEXT[],
  created_at TIMESTAMPTZ DEFAULT NOW()
);
CREATE TABLE IF NOT EXISTS primus_data (
  key TEXT PRIMARY KEY,
  value JSONB NOT NULL,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);
ALTER TABLE primus_logs ENABLE ROW LEVEL SECURITY;
ALTER TABLE primus_journal ENABLE ROW LEVEL SECURITY;
ALTER TABLE primus_data ENABLE ROW LEVEL SECURITY;
CREATE POLICY "allow_all_logs" ON primus_logs FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all_journal" ON primus_journal FOR ALL USING (true) WITH CHECK (true);
CREATE POLICY "allow_all_data" ON primus_data FOR ALL USING (true) WITH CHECK (true);
